547  CSMerkleNode *Insert(CSMerkleNode *root, const ChordKey &key,
548  const DataType &val)
549  {
550  if(root->IsLeaf()) {
551  return InsertLeaf(root, key, val);
552  }
553 
554 
555  if(root->left_->IsLeaf() && root->left_->key_ == key) {
556  root->left_ = Insert(root->left_, key, val);
557  return new CSMerkleNode(root->left_, root->right_);
558  } else if(root->right_->IsLeaf() && root->right_->key_ == key) {
559  root->right_ = Insert(root->right_, key, val);
560  return new CSMerkleNode(root->left_, root->right_);
561  }
562 
563  // To ensure predictable logarithmic lookups, we determine where to place
564  // a node based on the "distance" (floor(log2(key1 ^ key2))) of the key
565  // in question from the left and right keys.
566  KeyDist l_dist = Distance(key, root->left_->key_),
567  r_dist = Distance(key, root->right_->key_);
568 
569  // If the distance between left and right branches are equal, choose the
570  // branch w/ the lower hash.
571  if(l_dist == r_dist) {
572  auto *new_node = new CSMerkleNode(key, val);
573  ChordKey min_key = left_->key_ < right_->key_ ? left_->key_ :
574  right_->key_;
575  if(key < min_key) {
576  return new CSMerkleNode(new_node, root);
577  } else {
578  return new CSMerkleNode(root, new_node);
579  }
580  }
581 
582  // Otherwise, insert in the branch with the lower distance
583  if(l_dist < r_dist) {
584  root->left_ = Insert(root->left_, key, val);
585  } else {
586  root->right_ = Insert(root->right_, key, val);
587  }
588 
589  return new CSMerkleNode(root->left_, root->right_);
590  }

601  static CSMerkleNode *InsertLeaf(CSMerkleNode *leaf, const ChordKey &key,
602  const DataType &val)
603  {
604  // Take a leaf node, return a new node (to replace the leaf node),
605  // with the leaf node and a new node (constructed from key) as children.
606  if(leaf->key_ == key) {
607  leaf->value_ = val;
608  return leaf;
609  }
610 
611  auto *new_leaf = new CSMerkleNode(key, val);
612  if(key < leaf->key_) {
613  return new CSMerkleNode(new_leaf, leaf);
614  } else if(key > leaf->key_) {
615  return new CSMerkleNode(leaf, new_leaf);
616  } else {
617  return leaf;
618  }

628  DataType Lookup(CSMerkleNode *root, const ChordKey &key) const
629  {
630  // To see if the three contains a given key, we retrace the steps
631  // taken by the insertion algorithm until we hit a leaf. This leaf
632  // will be the location of the key if it exists inside the tree.
633  if(root->IsLeaf() && root->key_ == key) {
634  return root->value_.value();
635  }
636 
637  if(root->left_->IsLeaf() && root->left_->key_ == key) {
638  return root->left_->value_.value();
639  }
640 
641  if(root->right_->IsLeaf() && root->right_->key_ == key) {
642  return root->right_->value_.value();
643  }
644 
645  KeyDist l_dist = Distance(key, root->left_->key_),
646  r_dist = Distance(key, root->right_->key_);
647 
648  if(l_dist < r_dist) {
649  return Lookup(root->left_, key);
650  } else if(r_dist < l_dist) {
651  return Lookup(root->right_, key);
652  } else {
653  throw std::runtime_error("Value not in tree");
654  }
655  }

665  KvMap ReadRange(CSMerkleNode *root, const ChordKey &lower_bound,
666  const ChordKey &upper_bound)
667  {
668  KvMap results;
669 
670  if(root->IsLeaf()) {
671  if(root->key_.InBetween(lower_bound, upper_bound, true)) {
672  return {{ root->key_, root->value_.value() }};
673  } else {
674  return {{}};
675  }
676  }
677 
678  // The right node's keys are all greater than the left node's, so, if
679  // the max key of the left child is less than the lower_bound, we need
680  // to query the left side.
681  if(lower_bound <= root->left_->key_) {
682  bool left_leaf_in_range = root->left_->key_.InBetween(lower_bound,
683  upper_bound,
684  true);
685 
686  if(root->left_->IsLeaf() && left_leaf_in_range) {
687  results.insert({ root->left_->key_,
688  root->left_->value_.value() });
689  } else {
690  KvMap subtree_results = ReadRange(root->left_, lower_bound,
691  upper_bound);
692  results.insert(subtree_results.begin(), subtree_results.end());
693  }
694  }
695 
696  // The left node's keys are all less than the right node's keys.
697  // This means that we only need to query the right node if the left
698  // node's upper bound is less than the query's upper bound.
699  if(root->left_->key_ <= upper_bound) {
700  bool right_leaf_in_range = root->right_->key_.InBetween(lower_bound,
701  upper_bound,
702  true);
703 
704  if(root->right_->IsLeaf() && right_leaf_in_range) {
705  results.insert({ root->right_->key_,
706  root->right_->value_.value() });
707  } else {
708  KvMap subtree_results = ReadRange(root->right_,
709  root->left_->key_,
710  upper_bound);
711  results.insert(subtree_results.begin(), subtree_results.end());
712  }
713  }
714 
715  return results;
716  }

725  CSMerkleNode *Update(CSMerkleNode *root, const ChordKey &key,
726  const DataType &new_val)
727  {
728  // We've been through this already. Descend the tree while minimizing
729  // distance between key and node, update the leaf node if it exists.
730  if(root->IsLeaf()) {
731  if(root_->key_ == key) {
732  return new CSMerkleNode<DataType>(key, new_val);
733  } else {
734  return root;
735  }
736  }
737 
738  if(root->left_->IsLeaf() && root->left_->key_ == key) {
739  root->left_ = Update(root->left_, key, new_val);
740  return new CSMerkleNode(root->left_, root->right_);
741  }
742  if(root->right_->IsLeaf() && root_->right_->key_ == key) {
743  root->right_ = Update(root->right_, key, new_val);
744  return new CSMerkleNode(root->left_, root->right_);
745  }
746 
747  KeyDist l_dist = Distance(key, root->left_->key_),
748  r_dist = Distance(key, root->right_->key_);
749 
750  if(l_dist == r_dist) {
751  return root;
752  } else if(l_dist < r_dist) {
753  root->left_ = Update(root->left_, key, new_val);
754  } else {
755  root->right_ = Update(root->right_, key, new_val);
756  }
757 
758  return new CSMerkleNode(root->left_, root->right_);

768  CSMerkleNode *Delete(CSMerkleNode *root, const ChordKey &key)
769  {
770  // Deletes work essentially the same as insertion.
771  if(root->IsLeaf()) {
772  if(root_->key_ == key) {
773  return nullptr;
774  } else {
775  return root;
776  }

783  }
784

786  return root->left_;

812  CSMerkleNode *Next(CSMerkleNode *root, const ChordKey &key) const
813  {
814  if(root->IsLeaf()) {
815  return nullptr;
816  }
817 
818  // If left node is key, then its right sibling is necessarily the next
819  // greatest.
820  if(root->left_->IsLeaf() && root->left_->key_ == key) {
821  return root->right_;
822  }
823 
824  // If the rightmost node in the subtree contains the key, then there
825  // is no greater key in the subtree.
826  if(root->right_->IsLeaf() && root->right_->key_ == key) {
827  return nullptr;
828  }
829 
830  // If the key is the max key in the left subtree, then the next greatest
831  // key is in the right subtree.
832  if(root->left_->key_ <= key) {
833  return Next(root->right_, key);
834  }
835 
836  // Otherwise, it will be in the left subtree.
837  return Next(root->left_, key);
