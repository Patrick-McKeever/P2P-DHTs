108

110  // in order to aid lookups.
111  if(largest_key_.has_value() && kv_pair.first > largest_key_ ||
112  ! largest_key_.has_value())

168  KvMap ReadRange(const ChordKey &lower_bound, const ChordKey &upper_bound)
169  {
170  KvMap keys_in_range;
171 
172  // If this node is a leaf, return all kvs it holds within the specified
173  // range.
174  if(IsLeaf()) {
175  for(const auto &[key, val] : data_) {
176  if(key.InBetween(lower_bound, upper_bound, true)) {
177  keys_in_range.insert({ key, val });
178  }
179  }
180 
181  return keys_in_range;
182  }
183 
184  // Otherwise, determine the child which holds the upper bound and the
185  // child which holds the lower bound.
186  unsigned long lb_index = ChildNum(lower_bound),
187  ub_index = ChildNum(upper_bound);
188 
189  // Read all leaves of children in between the child containing the
190  // lower bound and the child containing the upper bound.
191  if(lb_index < ub_index) {
192  for(int i = (int) lb_index; i <= ub_index; ++i) {
193  MerkleTree nth_child = child_nodes_.at(i);
194  ChordKey lower = lower_bound < nth_child.GetMinKey() ?
195  nth_child.GetMinKey() : lower_bound;
196  ChordKey upper = upper_bound > nth_child.GetMaxKey() /*- 1*/ ?
197  nth_child.GetMaxKey() /*- 1*/ : upper_bound;
198 
199  KvMap keys_in_node = nth_child.ReadRange(lower, upper);
200  keys_in_range.insert(keys_in_node.begin(), keys_in_node.end());
201  }
202 
203  return keys_in_range;
204  }
205 
206  // If the lower bound is less than the upper bound (note that ChordKey
207  // is intended to represent a circular keyspace), then return the ranges
208  // [lb, MAX_KEY] and [0, ub].
209  else if(lb_index > ub_index) {
210  ChordKey max_key(mp::pow(mp::cpp_int(ChordKey::Base()),
211  ChordKey::Size()) - 1);
212  KvMap below_ub = ReadRange(ChordKey(0), upper_bound),
213  above_lb = ReadRange(lower_bound, max_key);
214  below_ub.insert(above_lb.begin(), above_lb.end());
215  return below_ub;
216  }
217 
218  return child_nodes_.at(lb_index).ReadRange(lower_bound, upper_bound);
219  }

280  std::optional<KvPair> Next(const ChordKey &key) const
281  {
282  // If the tree is totally empty, then there is no "next" key to return.
283  if(hash_ == ChordKey(0)) {
284  return std::nullopt;
285  }
286 
287  // If this is the root, and the key is larger than the largest key,
288  // the "next" key is the smallest key in the tree, because we treat
289  // keyspace as a logical ring.
290  if(key >= largest_key_ && position_.empty()) {
291  return GetSmallestEntry();
292  }
293 
294  if(IsLeaf()) {
295  // If this is a leaf, iterate through keys until we find one larger
296  // than the specified bound.
297  for(auto it = data_.begin(); it != data_.end(); ++it) {
298  if(it->first > key) {
299  return std::make_optional(*it);
300  }
301  }
302 
303  // If a larger key does not exist, then there is no "next" key in
304  // this (sub)tree.
305  return std::nullopt;
306  }
307 
308 
309  unsigned long node_ind = ChildNum(key);
310 
311  // It only makes sense to loop around if this node is the root.
312  for(unsigned long i = node_ind; i < num_children_; ++i)
313  {
314  std::optional<KvPair> next = child_nodes_.at(i).Next(key);
315  if(next.has_value()) {
316  return next;
317  }
318  }
319 
320  return std::nullopt;
321  }

705  {
706  if(key >= max_key_) {
707  return num_children_ - 1;
708  }

716  ChordKey::Size());

724  void Rehash()
725  {
726  std::string concatenated_keys;
727  if(IsLeaf()) {
728  if(data_.empty()) {
729  hash_ = ChordKey (0);
730  return;

732 
733  for(const auto &[key, _] : data_) {
734  concatenated_keys += std::string(key);
735  }

739  concatenated_keys += node.GetHash();

741

743  hash_ = ChordKey(0);
744  return;

755  void CreateChildren()
756  {
757  mp::uint256_t key_range(max_key_ - min_key_),
758  last_key = min_key_;
759  for(int i = 0; i < num_children_; ++i) {
760  mp::uint256_t ub = last_key + (key_range / num_children_);
761  std::deque<int> child_pos = position_;
762  child_pos.push_back(i);
763 
764  child_nodes_.emplace_back(ChordKey(last_key), ChordKey(ub),
765  child_pos);
766 
767  // Note that data_ is a map of ChordKeys and therefore ordered with
768  // increasing keys.
769  while(data_.begin() != data_.end() &&
770  data_.begin()->first.InBetween(last_key, ub - 1, true))
771  {
772  child_nodes_.back().data_.insert(*data_.begin());
773  data_.erase(data_.begin());
774  }
775  child_nodes_.back().Rehash();
776 
777  last_key = ub;
778  }
779  }
