128  void HandleRead(error_code ec, std::size_t bytes_xfrd)
129  {
130  // EC of 2 isn't really an "error", per se. Just means that the client
131  // closed the connection. The data is still available, and we can still
132  // work with it.
133  if(ec && ec.value() != 2) {
134  std::cerr << ec.message() << std::endl;
135  return;
136  }
137 
138  JSONCPP_STRING parse_err;
139  Json::Value json_req, json_resp;
140  // Changed
141  std::string client_req_str(data_);
142 
143  if (reader_->parse(client_req_str.c_str(),
144  client_req_str.c_str() + client_req_str.length(),
145  &json_req, &parse_err))
146  {
147  // If logging is enabled, log this request inside our queue.
148  if(logging_enabled_) {
149  request_log_->PushBack(json_req);
150  }
151 
152  try {
153  // Get JSON response.
154  json_resp = ProcessRequest(json_req);
155  json_resp["SUCCESS"] = true;
156  } catch (const std::exception &ex) {
157  // If ProcessRequest threw an error.
158  json_resp["SUCCESS"] = false;
159  json_resp["ERRORS"] = std::string(ex.what());
160  }
161  } else {
162  // If json parsing failed.
163  json_resp["SUCCESS"] = false;
164  json_resp["ERRORS"] = std::string(parse_err);
165  }
166 
167  reply_ = Json::writeString(writer_, json_resp);
168  auto self(this->shared_from_this());
169  async_write(socket_, buffer(reply_),
170  [this, self](error_code ec, std::size_t bytes_xfrd) {
171  HandleWrite(ec);
172  });
173  }

194  Json::Value ProcessRequest(Json::Value request)
195  {
196  Json::Value response;
197  std::string command = request["COMMAND"].asString();
198 
199  // If command is not valid, give a response with an error.
200  if(commands_.find(command) == commands_.end()) {
201  throw std::runtime_error("Invalid command.");
202  }
203 
204  // Otherwise, run the relevant handler.
205  else {
206  ReqHandlerType handler = commands_.at(command);
207  response = handler(request);
208  }
209  return response;
210  }
